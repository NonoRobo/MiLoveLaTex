    % #2 : légende
    % #3 : angle du scotch

        \fill[polaroidframe, drop shadow={shadow xshift=1.5mm, shadow yshift=-1.5mm, fill=black, opacity=0.65}] (0,0) rectangle (6,7);
        \node[anchor=south west, inner sep=0pt] at (0.5,1.5) {\includegraphics[width=5cm, height=5cm]{#1}};
